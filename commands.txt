create database std;
use std;
CREATE TABLE std (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(20) NOT NULL,
    email VARCHAR(20) UNIQUE NOT NULL,
    age INT,
    city VARCHAR(10),
    score DECIMAL(5,2) DEFAULT 0.00,
    enroll DATE,
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
